  function downloadHTML() {
    const htmlContent = document.documentElement.outerHTML;
    const blob = new Blob([htmlContent], { type: 'text/html' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'FIFA_2026_Bracket.html';
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  }

  function predictWithAI() {
    if (confirm("Use Trackr AI to predict the remaining upcoming matches?")) {
      // 1. Predict Semi-Finals (M101, M102) based on QF winners
      setPrediction('M101', 'France', 'England', 2, 1);

      championBox.innerText = 'France';
      championBox.classList.add('predicted');
      
      alert("AI Predictions applied! Upcoming matches have been filled.");
    }
  }

  function setPrediction(matchNo, team1, team2, score1, score2) {
    const matches = document.querySelectorAll('.match');
    matches.forEach(match => {
      const matchNoElement = match.querySelector('.match-no');
      if (matchNoElement && matchNoElement.innerText === matchNo) {
        const slots = match.querySelectorAll('.slot');
        const name1 = slots[0].querySelector('.team-name');
        const scoreEl1 = slots[0].querySelector('.score');
        name1.innerText = team1;
        name1.classList.add('predicted');
        scoreEl1.innerText = score1;
        scoreEl1.classList.add('predicted-score');
        
        const name2 = slots[1].querySelector('.team-name');
        const scoreEl2 = slots[1].querySelector('.score');
        name2.innerText = team2;
        name2.classList.add('predicted');
        scoreEl2.innerText = score2;
        scoreEl2.classList.add('predicted-score');
        
        if (score1 > score2) {
            name1.classList.remove('predicted'); name1.classList.add('winner');
            scoreEl1.classList.remove('predicted-score'); scoreEl1.classList.add('winner-score');
        } else {
            name2.classList.remove('predicted'); name2.classList.add('winner');
            scoreEl2.classList.remove('predicted-score'); scoreEl2.classList.add('winner-score');
        }
      }
    });
  }
